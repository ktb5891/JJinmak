create schema flaskdb;
use flaskdb;

drop table favorit_tbl;
create table if not exists favorit_tbl(
    favoriteNo int(3) auto_increment primary key,
    favoriteTitle varchar(30) not null,
    favoriteAddr varchar(50) not null,
    favoriteX float(12,7) not null,
    favoriteY float(11,7) not null,
    userID varchar(12) not null
);
-- drop table membertbl;
create table if not exists membertbl(
    userNo int(3) not null unique auto_increment,
    userID varchar(12) primary key,
    userName varchar(12) not null,
    pwd varchar(12) not null
);
